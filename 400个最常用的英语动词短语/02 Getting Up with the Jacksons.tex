\section{Getting Up with the Jacksons}

\begin{paracol}{2}

\englishtext{The alarm clock in Mr. and Mrs. Jackson's bedroom \ns{goes off} at 6:30 A.M. Sleepily, Mr. Jackson reaches toward it to \ns{turn it off}. He \ns{gives up} trying to reach it and lets it \ns{run down}.}

\switchcolumn

\nse{go off}{}{失去对…的兴趣;爆炸，开火;（电器设备）停止运转;进行，进展;变质，腐败;}
\nse{turn off}{}{（把…）关掉;完成;解雇;转向;}
\nse{give up}{}{放弃；投降;}
\nse{run down}{}{撞倒;走下坡路;诽谤;（使）虚弱;}

\switchcolumn*

\englishtext{By now Mrs. Jackson is awake. She \ns{turns on} the light beside the bed and shakes her husband. "It's time to \ns{get up}, George," she tells him.}

\switchcolumn

\nse{turn on}{}{打开（水、电视、收音机、灯、煤气等）;（使）感兴趣;（使）兴奋;发动;}
\nse{get up}{}{（使）起床;安排;举起;站起来;}

\switchcolumn*

\englishtext{"Aw-don't-" he mutters. But she \ns{keeps on} shaking him until he too is awake. He \ns{sits up}, but his eyes are still closed. Finally he \ns{stands up}.}

\switchcolumn

\nse{keep on}{}{继续雇用;继续前进;继续穿着;重复;}
\nse{sit up}{}{（使）端坐;晚睡;熬夜;<口>引起注意;}
\nse{stand up}{}{起立;竖立;经久耐用;<非正>失约;}

\switchcolumn*

\englishtext{Mrs. Jackson puts on a bathrobe. Mr. Jackson \ns{goes in} to take a shower and, awake at last, goes downstairs to set the table. In the kitchen Mrs. Jackson \ns{plugs in} the coffee maker. She wants to boil eggs but remembers that they \ns{used up} the eggs yesterday. She puts bread into the toaster and waits for it to \ns{pop up}. Mr. Jackson puts cereal, milk, and juice on the table.}

\switchcolumn

\nse{go in}{}{进去; 进入;开始;攻击;}
\nse{plug in}{}{插上电源的插头;堵住…上的洞;}
\nse{use up}{}{vt.& vi. 用完，耗尽;用光;}
\nse{pop up}{}{突然出现，冒出来;}

\switchcolumn*

\englishtext{Then Mrs. Jackson \ns{lets out} the cat. She calls the sixteen-year-old twins. "Breakfast!" she shouts. "Jim! Jane!"} 

\switchcolumn

\nse{let out}{}{使 （水、空气等） 通过;发出 （声音）;加宽 （衣服）;泄露;}

\switchcolumn*

\englishtext{In a few minutes all four are at the table. "What in the world do you \ns{have on}?" Jim asks Jane.}

\switchcolumn

\nse{have on}{hæv ɑn}{穿着;戴着;计划好;约定;}

\switchcolumn*

\englishtext{"I forgot to \ns{wash out} my clothes last night, so this morning I \ns{hunted up} these things to wear."}

\switchcolumn

\nse{wash out}{}{v. 破产，淘汰;刷洗;冲掉;}
\nse{hunt up}{}{（在报纸上、书上等）找到（资料等）;}

\switchcolumn*

\englishtext{"I thought you had \ns{given those old things away} long ago," Jim tells her. "They look shabby."}


\switchcolumn

\nse{give sth. away}{}{泄露;失去;赠送;（在婚礼仪式中）将新娘交给新郎;}

\switchcolumn*


\englishtext{"Leave me alone!’ she says angrily.}

\switchcolumn

\switchcolumn*

\englishtext{"Children," Mrs. Jackson begs, "please don't \ns{get into} a fight. I'm \ns{running out of} patience with both of you. I don't \ns{feel like} listening to another argument. Tom did not often argue before he left for college. Jane, you should \ns{put your clothes away}. Then you could find something better to wear. I've told both of you again and again, 'Please \ns{hang up} your clothes.'"}

\switchcolumn

\nse{get into}{}{进入;陷入;养成;}
\nse{run out of}{}{v. 用完;耗尽;从…跑出;}
\nse{feel like}{}{摸起来像是…，有…的感觉;想要…;}
\nse{put sth. away}{}{收起来; 放好;储存;打消;}
\nse{hung up}{}{挂上; 挂住;使等候;将电话筒挂上;}

\switchcolumn*

\englishtext{George Jackson is reading the newspaper. He laughs softly. "If you \ns{come across} something funny or even interesting, you might \ns{let us in on} it," his wife says.}

\switchcolumn

\nse{come across}{}{偶然发现，偶然遇见;}
\nse{let sb. in on}{}{<非正>让…知道[参与];}

\switchcolumn*

\englishtext{"I'm just \ns{looking at} the sports pages, Edna," he tells her. "You usually don’t \ns{care for} professional sports.}

\switchcolumn

\nse{look at}{}{查看;思考;考虑;研究;}
\nse{care for}{}{喜欢;照料;愿意;}

\switchcolumn*

\englishtext{She goes to \ns{see about} the cat and \ns{lets it in}. The twins \ns{pick up} their schoolbooks and go to school. George \ns{puts his coat on}. As he \ns{sets out} for work, he says to his wife, "Have a good day, Edna."}

\switchcolumn

\nse{see about}{}{查看;处理;考虑;}
\nse{let sth. in}{}{容许;加进;让…进入;把…降下;}
\nse{pick up}{}{拿起;提起;捡起;取走;学会;（身体）恢复，好转;}
\nse{put on}{}{表演(节目)，举办，展出，增加(体重);}
\nse{set out}{}{出发；启程；（怀着目标）开始工作;}

\switchcolumn*

\end{paracol}
